function calculateComplexity(n) {
  let result = 0;
  // Integral result is n^3 / 3
  let bound = Math.pow(n, 3) / 3;

  for (let i = 0; i < bound; i++) {
    for (let j = 0; j < n; j++) {
       result += i * j;
    }
  }
  return result;
}
